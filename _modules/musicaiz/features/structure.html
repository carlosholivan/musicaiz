import ruptures as rpt
import numpy as np
import networkx as nx
from dataclasses import dataclass

from musicaiz.features import (
    get_novelty_func,
    musa_to_graph
)


LEVELS = ["high", "mid", "low"]


@dataclass
class PeltArgs:
    penalty: int
    model: str
    minsize: int
    jump: int


class StructurePelt(PeltArgs):

    def __init__(self, level: str):
        if level == "high":
            self.args = PeltArgs(
                penalty=4,
                model="rbf",
                minsize=500,
                jump=2,
            )
        elif level == "mid":
            self.args = PeltArgs(
                penalty=4,
                model="rbf",
                minsize=100,
                jump=20,
            )
        elif level == "low":
            self.args = PeltArgs(
                penalty=4,
                model="rbf",
                minsize=5,
                jump=2,
            )
        else:
            raise ValueError(f"level {level} must be one of the following: {LEVELS}")


UNITS = ["ticks", "seconds", "beats", "bars", "notes"]


def get_structure_boundaries(
    musa_obj,
    level: str,
    unit: str
):
    """
    Get the beat indexes where a section ends.
    Uses get_structure_boundaries_notes function.
    """

    result = get_structure_boundaries_notes(musa_obj, level)
    bars = [musa_obj.notes[n].bar_idx for n in result]
    return bars

[docs]def get_structure_boundaries_notes(musa_obj, level: str):
    """
    Get the note indexes where a section ends.
    """
    st = StructurePelt(level)
    g = musa_to_graph(musa_obj)
    mat = nx.attr_matrix(g)[0]
    n = get_novelty_func(mat)
    nn = np.reshape(n, (n.size, 1))
    # detection
    algo = rpt.Pelt(
        model=st.args.model,
        min_size=st.args.minsize,
        jump=st.args.jump
    ).fit(nn)
    result = algo.predict(pen=st.args.penalty)
    return result
